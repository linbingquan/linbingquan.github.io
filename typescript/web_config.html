    "target": "es5" // 编译结果

# webpack.config.js

const path = require("path");

module.exports = {
  entry: "./src/index.ts", // 编译入口
  output: {
    path: path.resolve(__dirname, "dist"), // 编译输出
    filename: "bundle.js" // 打包文件名
  },
  module: {
    rules: [{ test: /\.ts$/, use: "ts-loader" }] // 加载TS文件
  },
  resolve: {
    extensions: [".ts"] // 查找TS文件
  },
};
